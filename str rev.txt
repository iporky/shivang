
public class strrev {

void reverse(String s,String p)
{
	int l= s.length();
	
	if(l==1)
	{
		p= p+ s.charAt(l-1);
		System.out.print(p);
	}
	else
	{
		p= p+ s.charAt(l-1);
		 
	      reverse(s.substring(0,l-1),p);
	}
}
public static void main(String[] args) {
   strrev a=new strrev();
   a.reverse("shivang","");
 }
}
